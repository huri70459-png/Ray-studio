def greet(name: str) -> str:
    """Greets a person by name."""
    return f"hello {name}"


def add(a: int, b: int) -> int:
    return a + b


class Counter:
    """Simple counter class."""

    def __init__(self) -> None:
        self.value = 0

    def inc(self) -> int:
        """Increment the counter."""
        self.value += 1
        return self.value
